# See all running containers and their health
docker compose ps

# View live logs from ALL containers (Ctrl+C to stop)
docker compose logs -f

# View logs for a specific container only
docker compose logs -f healthcare_server
docker compose logs -f healthcare_client
docker compose logs -f healthcare_mongodb


2. After Code Changes — Rebuild & Restart

# Rebuild images and restart all containers (most common)
docker compose up -d --build

# Rebuild only the server (faster if only backend changed)
docker compose up -d --build server

# Rebuild only the client (faster if only frontend changed)
docker compose up -d --build client


3. Force Full Reset (if containers are stale/cached)

# Stop all containers, remove them, rebuild from scratch
docker compose down && docker compose up -d --build

# Nuclear option: remove volumes too (WARNING: clears MongoDB data)
docker compose down -v && docker compose up -d --build


# Nuclear Reset (removes volumes too — ⚠️ clears local MongoDB data)

docker compose down -v; docker compose up -d --build
